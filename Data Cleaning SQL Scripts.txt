SELECT a.ParcelID, a.PropertyAddress, b.ParcelID, b.PropertyAddress, ISNULL(a.PropertyAddress, b.PropertyAddress)
FROM PortfolioProject.dbo.NashvilleHousing a
JOIN PortfolioProject.dbo.NashvilleHousing b
	on a.ParcelID = b.ParcelID
	AND a.UniqueID <> b.UniqueID
WHERE a.PropertyAddress IS NULL

UPDATE a
SET a.PropertyAddress = ISNULL(a.PropertyAddress, b.PropertyAddress)
FROM PortfolioProject.dbo.NashvilleHousing a
JOIN PortfolioProject.dbo.NashvilleHousing b
	on a.ParcelID = b.ParcelID
	AND a.UniqueID <> b.UniqueID
WHERE a.PropertyAddress IS NULL

SELECT SoldAsVacant,
	CASE 
	when SoldAsVacant = '0' THEN CAST('No' as int)
	when SoldAsVacant = '1' THEN CAST('Yes' as int)
	ELSE SoldAsVacant
	END AS SoldAsVacant
FROM PortfolioProject.dbo.NashvilleHousing

ALTER TABLE PortfolioProject.dbo.NashvilleHousing
Add SoldAsVacant2 Nvarchar(255);

ALTER TABLE PortfolioProject.dbo.NashvilleHousing
ALTER COLUMN SoldAsVacant2 Nvarchar(255);

UPDATE PortfolioProject.dbo.NashvilleHousing
SET SoldAsVacant2 = CASE
					when SoldAsVacant = 0 THEN 'No'
					when SoldAsVacant = 1 THEN 'Yes'
					ELSE NULL
					END

SELECT SoldAsVacant, SoldAsVacant2
FROM PortfolioProject.dbo.NashvilleHousing

SELECT DISTINCT(SoldAsVacant2), COUNT(SoldAsVacant2)
FROM PortfolioProject.dbo.NashvilleHousing
GROUP BY SoldAsVacant2

ALTER TABLE PortfolioProject.dbo.NashvilleHousing
ALTER COLUMN SoldAsVacant Nvarchar(255);

UPDATE PortfolioProject.dbo.NashvilleHousing
SET SoldAsVacant = SoldAsVacant2

SELECT DISTINCT(SoldAsVacant), COUNT(SoldAsVacant)
FROM PortfolioProject.dbo.NashvilleHousing
GROUP BY SoldAsVacant
ORDER BY SoldAsVacant

ALTER TABLE PortfolioProject.dbo.NashvilleHousing
DROP COLUMN SoldAsVacant2

SELECT *,
	ROW_NUMBER() OVER (
	PARTITION BY ParcelID,
					PropertyAddress,
					SalePrice,
					SaleDate,
					LegalReference
					ORDER BY
					UniqueID
					) Row_num
FROM PortfolioProject.dbo.NashvilleHousing
--ORDER BY ParcelID
)
SELECT *
FROM RowNumCTE
WHERE Row_num > 1
ORDER BY PropertyAddress
